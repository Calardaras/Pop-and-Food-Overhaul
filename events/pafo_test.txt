pafo_test.3 = {
	#add
	type = country_event
	title = pafo_test.3.t
	desc = pafo_test.3.desc
	picture = aqueducts
	hidden = no
	immediate = {
		every_commander = {
			limit = { 
				OR = {
					is_governor = yes
					is_ruler = yes
				}
			}
			every_governor_state ={
				add_state_modifier = {
					name = levies_raised
					duration = -1
				}
			}
		}
	}
	option = {
		name = pafo_test.3.a
	}
}
