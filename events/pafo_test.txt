namespace = pafo_test
pafo_test.1 = {
	#initial
    type = country_event
	
	hidden = yes

	immediate = {
		every_province = {
			switch = {
				trigger = terrain
				plains = {
					set_variable = { name = fields value = 16 }

				}
				mountain = {
					set_variable = { name = fields value = 8 }
					set_variable = { name = curr_fields value = 0 }
				}
				hills = {
					set_variable = { name = fields value = 12 }
					set_variable = { name = curr_fields value = 0 }
				}
				desert = {
					set_variable = { name = fields value = 8 }
					set_variable = { name = curr_fields value = 0 }
				}
				marsh = {
					set_variable = { name = fields value = 10 }
					set_variable = { name = curr_fields value = 0 }
				}
				jungle = {
					set_variable = { name = fields value = 12 }
					set_variable = { name = curr_fields value = 0 }
				}
				farmland = {
					set_variable = { name = fields value = 20 }
					set_variable = { name = curr_fields value = 0 }
				}
				forest = {
					set_variable = { name = fields value = 10 }
					set_variable = { name = curr_fields value = 0 }
				}
			}
		}
	}
	option = {
		name = pafo_test.1.a
	}
}

pafo_test.3 = {
	#add food penalty
	type = country_event
	title = pafo_test.3.t
	desc = pafo_test.3.desc
	picture = aqueducts
	hidden = no
	immediate = {
		every_commander = {
			limit = { 
				OR = {
					is_governor = yes
					is_ruler = yes
				}
			}
			every_governor_state ={
				add_state_modifier = {
					name = levies_raised
					duration = 365
				}
			}
		}
	}
	option = {
		name = pafo_test.3.a
	}
}

pafo_test.4 = {
	#clear food penalty
	type = country_event
	title = pafo_test.4.t
	desc = pafo_test.4.desc
	picture = aqueducts
	hidden = no
	immediate = {
		every_country_state = {
			limit = {
				governor_or_ruler = {
					is_general = no
				}
		    }
			remove_state_modifier = levies_raised
		}
	}
	option = {
		name = pafo_test.4.a
	}
}
